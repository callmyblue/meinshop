<div class="cart">
  <h3>Warenkorb</h3>
  <ul id="warenkorb"></ul>
  <p><strong>Gesamt:</strong> <span id="gesamt">0</span>€</p>
  <button onclick="bezahlen()">Mit PayPal bezahlen</button>
</div>

<script>
  const shop = document.getElementById("shop");
  const warenkorbList = document.getElementById("warenkorb");
  const gesamtAnzeige = document.getElementById("gesamt");
  let warenkorb = JSON.parse(localStorage.getItem("warenkorb")) || [];
  let gesamt = 0;

  async function ladeProdukte() {
    const res = await fetch("produkte.json");
    const produkte = await res.json();

    produkte.forEach(p => {
      const el = document.createElement("div");
      el.className = "product";
      el.innerHTML = `
        <img src="${p.bild}" alt="${p.name}">
        <h3>${p.name}</h3>
        <p>${p.preis}€</p>
        <button onclick='inWarenkorb(${JSON.stringify(p)})'>In den Warenkorb</button>
      `;
      shop.appendChild(el);
    });
    aktualisiereWarenkorb();
  }

  function inWarenkorb(p) {
    warenkorb.push(p);
    localStorage.setItem("warenkorb", JSON.stringify(warenkorb));
    aktualisiereWarenkorb();
  }

  function aktualisiereWarenkorb() {
    warenkorbList.innerHTML = "";
    gesamt = 0;
    warenkorb.forEach(p => {
      const li = document.createElement("li");
      li.textContent = ${p.name} - ${p.preis}€;
      warenkorbList.appendChild(li);
      gesamt += p.preis;
    });
    gesamtAnzeige.textContent = gesamt.toFixed(2);
  }

  function bezahlen() {
    const url = "https://www.paypal.com/paypalme/deinname/" + gesamt.toFixed(2);
    window.open(url, "_blank");
  }

  ladeProdukte();
</script>
